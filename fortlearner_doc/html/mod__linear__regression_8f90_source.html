303  real(kind=8), allocatable :: xy(:)

317 
318  ! Compute X^T X
319  call mattxmat(mat_sq, x, n_rows, n_cols, with_intercept=this%hparam%fit_intercept)
320 
321  ! Cholesky decomposition of X^T X
