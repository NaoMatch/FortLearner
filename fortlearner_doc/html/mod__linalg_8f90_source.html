146  implicit none

148  real(kind=8), intent(in) :: input_vector(n_cols)
149  real(kind=8), intent(inout) :: output_vector(n_samples)
150  integer(kind=8), intent(in) :: indices(n_samples)
151  integer(kind=8), intent(in) :: n_rows, n_cols, n_samples
152 
153  integer(kind=8) :: i, j, k, n_cols_unroll, idx
154  real(kind=8) :: tmp_out, tmp_input, buffer_input(cache_size_multi_mat_vec)
155  output_vector = 0
156 
157  n_cols_unroll = n_cols - mod(n_cols, cache_size_multi_mat_vec)
158  do j=1, n_cols_unroll, cache_size_multi_mat_vec
159  do k=0, cache_size_multi_mat_vec-1, 1
160  buffer_input(k+1) = input_vector(j+k)
161  end do
162 
163  do i=1, n_samples, 1
164  tmp_out = 0
165  idx = indices(i)
166  do k=0, cache_size_multi_mat_vec-1, 1
167  tmp_out = tmp_out + matrix(idx,j+k) * buffer_input(k+1)
168  end do
169  output_vector(i) = output_vector(i) + tmp_out
170  end do
171  end do
172 
173  do j=n_cols_unroll+1, n_cols, 1
174  tmp_input = input_vector(j)
175  do i=1, n_samples, 1
176  idx = indices(i)
177  output_vector(i) = output_vector(i) + matrix(idx,j) * tmp_input
178  end do
179  end do
